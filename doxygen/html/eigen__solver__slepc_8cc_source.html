111 
112  // Create the "Shell" Matrix

114  ierr = MatCreateShell (PETSC_COMM_WORLD, n_dofs, n_dofs, n_dofs, n_dofs,

116  ierr = MatShellSetOperation (shell_mat, MATOP_MULT, (void

119  ierr = EPSSetOperators (eps, shell_mat, NULL);
120  ierr = EPSSetProblemType (eps, EPS_NHEP); // The problem is non-symmetric
121  ierr = EPSSetTolerances (eps, tol_eps, max_iterations_eps);
122  ierr = EPSSetWhichEigenpairs (eps, EPS_LARGEST_MAGNITUDE);
123  ierr = EPSSetType (eps, EPSKRYLOVSCHUR);

205  PetscErrorCode
206  shell_vmult (Mat shell_mat,
207  Vec src,
208  Vec dst)
209  {
210  PetscErrorCode ierr;
211 
212  // The context of the shell matrix is a pointer to the EigenSolverSLEPC object
213  // so we can access the data of the problem.
214  void *ctx;
215  ierr = MatShellGetContext (shell_mat, &ctx);
216  CHKERRQ(ierr);
217  EigenSolverSLEPC<dim> *EPSobject = (EigenSolverSLEPC<dim>*) ctx;
218  /* Copy to SnVector */
219  EPSobject->copy_to_sn_vector (src, EPSobject->sn_vector_tmp_src);
220  /* Apply the operator y = A x. */
221  EPSobject->m_eigen_prob->vmult (EPSobject->sn_vector_tmp_dst,
222  EPSobject->sn_vector_tmp_src);
223  /* Copy back to Petsc Vector */
224  EPSobject->copy_to_petsc_vector (EPSobject->sn_vector_tmp_dst, dst);
225 
226  EPSobject->n_matvec_multiplications++;
227  log_print_text (
228  "SLEPc Multiplication: " + num_to_string (
229  EPSobject->n_matvec_multiplications), 1);
230 
231  CHKERRQ(ierr);
232  return 0;
233  }
