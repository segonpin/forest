67  EigenSolver<dim>::power_iteration (SnVector & vec,

69  {
70  Timer timer;
71  timer.restart ();
72 
73  // Declare variables for the keff.
74  double keff_old, err_keff, err_vec;
75 
76  // Get the initial guess.
77  m_tmp_new = vec;
78  m_tmp_new *= 1.0 / m_tmp_new.l2_norm ();
79 
80  log_print_line ();
81  bool keff_converged = false;
82  for (unsigned int iter = 0; iter < m_max_it; ++iter)
83  {
84  // Move new (or guess) to old solution.
85  m_tmp_old = m_tmp_new;
86  keff_old = keff;
87 
88  // Apply the operator y = A x.
89  m_eigen_prob->vmult (m_tmp_new, m_tmp_old);
90 
91  // Update the k-effective and the eigenvector.
92  keff = m_tmp_new * m_tmp_old; //this->update_keff (m_tmp_new, m_tmp_old);
93  m_tmp_new *= 1.0 / m_tmp_new.l2_norm ();
94  m_tmp_old.add (-1.0, m_tmp_new);
95 
96  // estimate the errors
97  err_keff = std::abs (keff - keff_old) / std::abs (keff_old);
98  err_vec = m_tmp_old.l2_norm ();
99  const double error_comp = err_vec + err_keff;
100 
101  // Print information about the current mp_state of the convergence.
102  const double spg = m_eigen_prob->get_sweeps_per_group ();
103  keff_converged = log_table_keff_wgit (iter, keff, spg, error_comp, m_tol);
104  // keff_converged = log_table_keff (iter, keff, err, m_tol);
105 
106  // Stop the loop if convergence has been achieved.
107  if (keff_converged)
108  {
109  log_print_text (
110  "Total number of iterations: " + std::to_string (iter + 1), 1);
111  break;
112  }
113  }
114 
115  // Information about the number of sweeps and the convergence achieved.
116  // Move the solution to the container to be returned.
117  const unsigned int n_total_sweeps = m_eigen_prob->get_total_sweeps ();
118  log_print_text (
119  "Total number of sweeps: " + std::to_string (n_total_sweeps), 1);
120  this->convergence_msg (keff_converged);
121  vec = m_tmp_new;
122 
123  timer.stop ();
124  StatsTimer::instance ().add ("Power iteration", timer.wall_time ());
