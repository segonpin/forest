245  TransOpMatrixBuilt<dim>::trans_integrate_cell_term (DoFInfo &dinfo,
246  CellInfo &info,
247  const unsigned int g,
248  const Point<dim> & beta,
249  const double /* weight */)
250  {
251  const FEValuesBase<dim> &fe_v = info.fe_values ();
252  FullMatrix<double> &local_matrix = dinfo.matrix (0).matrix;
253  const std::vector<double> &JxW = fe_v.get_JxW_values ();
254  /*
255  * Here the sigmat should come from TH
256  */
257  const double sigmat = mp_state.mp_data.mp_mat.m_xs.at (
258  dinfo.cell->material_id ()).m_sigmat[g];
259  for (unsigned int q = 0; q < fe_v.n_quadrature_points; ++q)
260  {
261  for (unsigned int i = 0; i < fe_v.dofs_per_cell; ++i)
262  {
263  for (unsigned int j = 0; j < fe_v.dofs_per_cell; ++j)
264  {
265  local_matrix (i, j) += -beta * fe_v.shape_grad (i, q)
266  * fe_v.shape_value (j, q) * JxW[q]
267  + sigmat * fe_v.shape_value (i, q)
268  * fe_v.shape_value (j, q) * JxW[q];
269  }
