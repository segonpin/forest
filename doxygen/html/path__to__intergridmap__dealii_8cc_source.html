77  template <class GridClass>
78  void
79  InterGridMapSebas<GridClass>::make_mapping (const GridClass &source_grid,
80  const GridClass &destination_grid)
81  {
82  // first delete all contents
83  clear ();
84 
85  // next store pointers to grids
86  this->source_grid = &source_grid;
87  this->destination_grid = &destination_grid;
88 
89  // then set up the containers from
90  // scratch and fill them with end-iterators
91  const unsigned int n_levels = get_n_levels (source_grid);
92  mapping.resize (n_levels);
93  for (unsigned int level = 0; level < n_levels; ++level)
94  {
95  // first find out about the highest
96  // index used on this level. We could
97  // in principle ask the triangulation
98  // about this, but we would have to
99  // know the underlying data structure
100  // for this and we would like to
101  // avoid such knowledge here
102  unsigned int n_cells = 0;
103  cell_iterator cell = source_grid.begin (level), endc = source_grid.end (
104  level);
105  for (; cell != endc; ++cell)
106  {
107  if (static_cast<unsigned int> (cell->index ()) > n_cells)
108  {
109  n_cells = cell->index ();
110  }
111  }
112 
113  // note: n_cells is now the largest
114  // zero-based index, but we need the
115  // number of cells, which is one larger
116  mapping[level].resize (n_cells + 1, destination_grid.end ());
117  };
118 
119  // now make up the mapping
120  // loop over all cells and set the user
121  // pointers as well as the contents of
122  // the two arrays. note that the function
123  // takes a *reference* to the int and
124  // this may change it
125  cell_iterator src_cell = source_grid.begin (0), dst_cell =
126  destination_grid.begin (0), endc = source_grid.end (0);
127  for (; src_cell != endc; ++src_cell, ++dst_cell)
128  {
129  set_mapping (src_cell, dst_cell);
130  }
131 
132  // little assertion that the two grids
133  // are indeed related:
134  Assert(dst_cell == destination_grid.end(0), ExcIncompatibleGrids ());
135  }

275  {

277 #ifdef DEBUG

279 #endif

285 
286  Vector cache;
287 
288  // Looping over the finest common
289  // mesh, this means that source and
290  // destination cells have to be on the
291  // same level and at least one has to
292  // be active.
293  //
294  // Therefor, loop over all cells
295  // (active and inactive) of the source
296  // grid ..
297  typename DH<dim, spacedim>::cell_iterator cell1 = dof1.begin ();
298  const typename DH<dim, spacedim>::cell_iterator endc1 = dof1.end ();
299 
300  for (; cell1 != endc1; ++cell1)
301  {
302  const typename DH<dim, spacedim>::cell_iterator cell2 =
303  intergridmap[cell1];
304 
305  // .. and skip if source and destination
306  // cells are not on the same level ..
307  if (cell1->level () != cell2->level ())
308  {
309  continue;
310  }
311  // .. or none of them is active.
312  if (!cell1->active () && !cell2->active ())
313  {
314  continue;
315  }
316 
317  Assert(cell1->get_fe().get_name() == cell2->get_fe().get_name(),
318  ExcMessage ("Source and destination cells need to use the same finite element"));
319 
320  cache.reinit (cell1->get_fe ().dofs_per_cell);
321 
322  // Get and set the corresponding
323  // dof_values by interpolation.
324  cell1->get_interpolated_dof_values (u1, cache);
325  cell2->set_dof_values_by_interpolation (cache, u2);
326  }
327 
328  // Apply hanging node constraints.
329  constraints.distribute (u2);
