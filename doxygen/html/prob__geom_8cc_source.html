245  {

272  double dy = std::abs ((cell->center () - center)[1]);
273  double radius = real_pins[pin].get_radius () * std::sqrt (2.0) / 2.;
274  if (dx < radius and dy < radius)

276  cell->set_refine_flag (RefinementCase<dim>::cut_xy);
277  }

289  if (std::abs (orthogonality - 1.0) < 1.e-6)

291  direction = 1 - f / 2;
292  break;
293  }
294  }

308  }

324  }

348  void
349  ProbGeom<dim>::set_boundary_indicators ()
350  {
351  const double eps = 1.e-10;
352  for (typename Triangulation<dim, dim>::active_cell_iterator cell =
353  m_tria.begin_active (); cell != m_tria.end (); ++cell)
354  {
355  for (unsigned int f = 0; f < GeometryInfo<dim>::faces_per_cell; ++f)
356  {
357  if (cell->at_boundary (f))
358  {
359 
360  // we consider the center of the face and a vertex
361  const Point<dim> cell_center = cell->center ();
362  const Point<dim> face_center = cell->face (f)->center ();
363  const Point<dim> face_vertex = cell->face (f)->vertex (0);
364  // generate a vector parallel to the face
365  const Tensor<1, dim> face_parallel = face_center - face_vertex;
366  const Tensor<1, dim> face_normal = face_center - cell_center;
367  unsigned int normal_axis = -1;
368  for (unsigned int d = 0; d < dim; ++d)
369  if (std::abs (face_parallel[d]) < eps)
370  normal_axis = d;
371 
372  const double normal_component = face_normal[normal_axis];
373 
374  Assert(normal_axis == 0 or normal_axis == 1 or normal_axis == 2,
375  ExcMessage("Wrong axis position."));
376  Assert(std::abs (normal_component) >= 1.e-10,
377  ExcMessage("Wrong direction. Normal can not be zero."))
378  if (std::abs (normal_component) <= 1.e-10)
379  std::cout << "Wrong direction. Normal can not be zero."
380  << std::endl;
381 
382  // How we do number this?
383  // x- x+ y- y+ z- z+
384  // 0 1 2 3 4 5
385  unsigned int face_index = -1;
386  if (normal_component <= 0.0)
387  face_index = 2 * normal_axis;
388  else
389  face_index = 2 * normal_axis + 1;
390  cell->face (f)->set_boundary_id (face_index);
391  }
