635  EqTransport<dim>::set_rhs (BlockVector<double> & dst,
636  const BlockVector<double> &src) const
637  {
638  /* We need these constants. */
639  const unsigned int n_angles = m_ord->get_n_angles ();
640  const unsigned int n_leg_mom = mp_state.get_n_leg_mom ();
641 
642  /* Check if the energy group is out of range. */
643  Assert(m_g < mp_state.get_n_groups (),
644  ExcMessage("group index out of range."));
645  /* if we have more than 1 legendre moments this is wrong */
646  Assert(n_leg_mom == 1, ExcMessage("n_leg_mom different from 1."));
647 
648  /* Initialize destination vector to zero. */
649  dst = 0;
650  /* First we prepare the right hand side */
651  for (unsigned int n = 0; n < n_angles; ++n)
652  {
653  /* Moments to direction and stored in m_v2. */
654  m_M.vmult (m_v2, src, n);
655  /* Transport sweep for direction n */
656  m_L0->apply_L0_inv (m_v3, m_v2, m_g, n);
657  /* Get outgoing fluxes for reflective boundary conditions. */
658  m_BV->get_reflective (dst, m_v3, n, n_leg_mom);
659  /* Apply directions-to-moments operator to recover the scalar flux. */
660  m_D.vmult_add (dst, m_v3, n);
661  }
