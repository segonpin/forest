500  void
501  InputGeom::check_core () const
502  {
503  // Dimensions should be consistent
504 
505  unsigned int dim = m_dim;
506 
507  // more dimensions are not ready yet
508  Assert(dim == 1 or dim == 2 or dim == 3,
509  dealii::ExcMessage("dimension should be 1 or 2"));
510 
511  // First we check the dimensions with respect the number of nodes
512  for (unsigned int i = dim; i < m_core.m_n_nodes.size (); ++i)
513  Assert(1 == m_core.m_n_nodes[i],
514  dealii::ExcMessage("n_nodes for dimensions not used should be one"));
515 
516  // Now we chek that the number of nodes per direction agree with the lengths
517  for (unsigned int i = 0; i < dim; ++i)
518  Assert(m_core.m_length[i].size () == m_core.m_n_nodes[i],
519  dealii::ExcMessage("lengths vs n_nodes per axis error"));
520 
521  // Now we chek the number of nodes per axis versus the components
522  if (dim == 1)
523  {
524  Assert(m_core.m_indices[0][0].size () == m_core.m_n_nodes[0],
525  dealii::ExcMessage("lengths vs n_nodes per axis error"));
526  }
527  else
528  {
529  Assert(m_core.m_indices[0][0].size () == m_core.m_n_nodes[0],
530  dealii::ExcMessage("lengths vs n_nodes per axis error"));
531  for (unsigned int i = 0; i < m_core.m_indices[0].size (); ++i)
532  Assert(m_core.m_indices[0][i].size () == m_core.m_n_nodes[dim - 2],
533  dealii::ExcMessage("lengths vs n_nodes per axis error"));
534  }
535 
536  // Now the boundary conditions
537  Assert(m_core.m_bcs.size () == dim * 2,
538  dealii::ExcMessage("boundary vs dimension*2 error"));
539 
540  } //end InputGeom::check_core()

542  void
543  InputGeom::check_lattices () const
544  {
545  unsigned int dim = m_dim;
546 
547  for (typename std::map<unsigned int, Lattice>::const_iterator lattice_it =
548  m_lattices.begin (); lattice_it != m_lattices.end (); ++lattice_it)
549  {
550  const Lattice &lattice = lattice_it->second;
551 
552  // First we check the dimensions with respect the number of nodes
553  for (unsigned int i = dim; i < lattice.m_n_nodes.size (); ++i)
554  Assert(1 == lattice.m_n_nodes[i],
555  dealii::ExcMessage("n_nodes for dimensions not used should be one"));
556 
557  // Depending on the type of lattice, the number of elements or the water gap
558  // should be consistent
559  if (lattice.m_latt_type == LattType::GRID)
560  {
561  for (unsigned int i = 0; i < dim; ++i)
562  Assert(lattice.m_length[i].size () == lattice.m_n_nodes[i],
563  dealii::ExcMessage("lengths vs n_nodes per axis error"));
564  }
565  else if (lattice.m_latt_type == LattType::PINMAP)
566  {
567  /*Assert(lattice->water_gap.size () == 2 * dim,
568  dealii::ExcMessage("water_gap.size() vs dimension*2 error"));*/
569  }
570  else
571  {
572  Assert(false, dealii::ExcMessage("Latt_type not recognised"));
573  }
574 
575  // Now we chek the number of nodes per axis versus the components
576  if (dim == 1)
577  {
578  Assert(lattice.m_components.size () == dim,
579  dealii::ExcMessage("number of lines in components is not valid for 1D"));

581  dealii::ExcMessage("components vs n_nodes per axis error"));
582  }

586  dealii::ExcMessage("components vs n_nodes per axis error"));

589  {

591  dealii::ExcMessage("lattice: components vs n_nodes in axis " + Forest::num_to_string ( i) + " error\n"));
