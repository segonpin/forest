86  {
87  if (m_verbose)
88  std::cout << "start build_pin_power" << std::endl;
89 
90  // We need this constants
91  const unsigned int n_groups = data.mp_mat.get_n_groups ();
92  const unsigned int n_dofs = dof_handler.n_dofs();
93  const unsigned int n_cells = geom.m_tria.n_active_cells ();
94  const unsigned int n_pins = geom.get_n_pins ();
95  const unsigned int n_ass = geom.get_n_assemblies ();
96 
97  // Initialize all that is needed to iterate over dofs and cells
98  const unsigned int fe_degree = geom.mp_data.mp_settings.get_fe_degree ();
99  QGauss<dim> quadrature_formula (fe_degree + 1);
100  FEValues<dim> fe_values (fe, quadrature_formula,
101  update_values | update_quadrature_points | update_volume_elements
102  | update_JxW_values);
103  const unsigned int n_q_points = quadrature_formula.size ();
104  std::vector<unsigned int> global_dof_ind (fe.dofs_per_cell);
105 
106  // Now, let's multiply the scalar flux per group per the
107  // corresponding fission cross-section, and add them to get
108  // the power distribution.
109  for (typename DoFHandler<dim, dim>::active_cell_iterator cell_it =
110  dof_handler.begin_active (); cell_it != dof_handler.end (); ++cell_it)
111  {
112  fe_values.reinit (cell_it);
113  cell_it->get_dof_indices (global_dof_ind);
114  const unsigned int mat = cell_it->material_id ();
115  std::vector<double> nusigfission (n_groups, 0.);
116  if (data.mp_mat.m_xs.at (mat).exist_sigf)
117  {
118  nusigfission = data.mp_mat.m_xs.at (mat).m_sigf;
119  }
120  else if (data.mp_mat.m_xs.at (mat).exist_nusigf)
121  {
122  nusigfission = data.mp_mat.m_xs.at (mat).m_nusigf;
123  }
124  else
125  {
126  nusigfission.resize (n_groups, 0.);
127  }
128  for (unsigned int i = 0; i < fe.dofs_per_cell; ++i)
129  {
130  for (unsigned int g = 0; g < mp_state.get_n_groups (); ++g)
131  {
132  m_power_dofs[global_dof_ind[i]] += nusigfission[g]
133  * mp_state.m_scalar_flux.m_data[g].block (0)[global_dof_ind[i]];
134  }
135  }
136  }
137 
138  // We go from dof to cell -------------------------------------------------
139 
140  // how much is the weight of each pin?
141  m_weight_cell.resize (n_cells, 0.0);
142  for (typename DoFHandler<dim, dim>::active_cell_iterator cell_it =
143  dof_handler.begin_active (); cell_it != dof_handler.end (); ++cell_it)
144  {
145  const unsigned int cell = cell_it->user_index ();
146  m_weight_cell[cell] = cell_it->measure ();
147  }
148  // Now we get the integral of the flux and power over each cell
149  m_power_cell.resize (n_cells, 0.0);
150  m_flux_cell.resize (n_groups, std::vector<double> (n_cells, 0.0));
151  for (typename DoFHandler<dim, dim>::active_cell_iterator cell_it =
152  dof_handler.begin_active (); cell_it != dof_handler.end (); ++cell_it)
153  {
154  const unsigned int cell = cell_it->user_index ();
155  fe_values.reinit (cell_it);
156  cell_it->get_dof_indices (global_dof_ind);
157  for (unsigned int i = 0; i < fe.dofs_per_cell; ++i)
158  {
159  // Calculate the weight of this degree of freedom
160  double weight_dof_i = 0;
161  for (unsigned int q = 0; q < n_q_points; ++q)
162  {
163  weight_dof_i += fe_values.shape_value (i, q) * fe_values.JxW (q);
164  }
165  // add the corresponding contribution
166  m_power_cell[cell] += m_power_dofs[global_dof_ind[i]] * weight_dof_i;
167  for (unsigned int g = 0; g < mp_state.get_n_groups (); ++g)
168  {
169  m_flux_cell[g][cell] += weight_dof_i
170  * mp_state.m_scalar_flux.m_data[g].block (0)[global_dof_ind[i]];
171  }
172  }
173  }
174 
175  // We go from cell to pin -------------------------------------------------
176 
177  // how much is the weight of each pin?
178  m_weight_pin.resize (n_pins, 0);
179  for (typename DoFHandler<dim, dim>::active_cell_iterator cell_it =
180  dof_handler.begin_active (); cell_it != dof_handler.end (); ++cell_it)
181  {
182  const unsigned int cell = cell_it->user_index ();
183  const unsigned int pin = geom.get_cell2pin (cell);
184  m_weight_pin[pin] += m_weight_cell[cell];
185  }
186  // Now we get the flux and power over each pin
187  m_power_pin.resize (n_pins, 0);
188  m_flux_pin.resize (n_groups, std::vector<double> (n_pins, 0.0));
189  for (typename DoFHandler<dim, dim>::active_cell_iterator cell_it =
190  dof_handler.begin_active (); cell_it != dof_handler.end (); ++cell_it)
191  {
192  const unsigned int cell = cell_it->user_index ();
193  const unsigned int pin = geom.get_cell2pin (cell);
194  m_power_pin[pin] += m_power_cell[cell];
195  for (unsigned int g = 0; g < mp_state.get_n_groups (); ++g)
196  {
197  m_flux_pin[g][pin] += m_flux_cell[g][cell];
198  }
199  }
200 
201  // We go from pin to assembly ---------------------------------------------
205 
206  // how much is the weight of each assembly?
207  m_weight_ass.resize (n_ass, 0);
208  for (unsigned int pin = 0; pin < n_pins; ++pin)
209  {
210  const unsigned int ass = geom.get_pin2assembly (pin);
211  m_weight_ass[ass] += m_weight_pin[pin];
212  }
213  // Now we get the flux and power over each assembly
214  m_power_ass.resize (n_ass, 0);
215  m_flux_ass.resize (n_groups, std::vector<double> (n_ass, 0.0));
216  for (unsigned int pin = 0; pin < n_pins; ++pin)
217  {
218  const unsigned int ass = geom.get_pin2assembly (pin);
219  m_power_ass[ass] += m_power_pin[pin];
220  for (unsigned int g = 0; g < mp_state.get_n_groups (); ++g)
221  {
222  m_flux_ass[g][ass] += m_flux_pin[g][pin];
223  }
224  }
225 
226  // We go from assembly to total---------------------------------------------
227  double power_total = 0.0;
228  double weight_total = 0.0;
229  for (unsigned int i = 0; i < m_power_ass.size (); ++i)
230  {
231  weight_total += m_weight_ass[i];
232  power_total += m_power_ass[i];
233  }
234 
235  // ------------------------------------------------------------------------
236  // normalization ----------------------------------------------------------
237  // pow*w' / sum(w)
238  m_scaling = 1. / (power_total/weight_total);
239 
240 
241  // Normalization (average power per dofs is equal to one) ------------------
242  m_norm_power_dofs.reinit (n_dofs, 0.0);
243  m_norm_flux_dofs.resize (n_groups, m_norm_power_dofs);
244  for (unsigned int i = 0; i < m_power_dofs.size (); ++i)
245  {
246  m_norm_power_dofs[i] = m_power_dofs[i]*m_scaling;
247  }
248  // Normalization (average flux per ass is equal to one (for fast group))
249  for (unsigned int g = 0; g < n_groups; ++g)
250  {
251  for (unsigned int i = 0; i < n_dofs; ++i)
252  {
253  m_norm_flux_dofs[g][i] = mp_state.m_scalar_flux.m_data[g].block (0)[i]
254  * m_scaling;
255  }
256  }
257 
258  // Normalization (average power per cell is equal to one) ------------------
259  m_norm_power_cell.resize (n_cells, 0.0);
260  m_norm_flux_cell.resize (n_groups, std::vector<double> (n_cells, 0.0));
261  for (unsigned int i = 0; i < m_power_cell.size (); ++i)
262  {
263  m_norm_power_cell[i] = m_power_cell[i]/m_weight_cell[i]*m_scaling;
264  }
265  // Normalization (average flux per ass is equal to one (for fast group))
266  for (unsigned int g = 0; g < m_flux_cell.size (); ++g)
267  {
268  for (unsigned int i = 0; i < m_flux_cell[g].size (); ++i)
269  {
270  m_norm_flux_cell[g][i] = m_flux_cell[g][i] / m_weight_cell[i]
271  * m_scaling;
272  }
273  }
274 
275  // Normalization (average power per pin is equal to one) ------------------
276  m_norm_power_pin.resize (n_pins, 0);
277  m_norm_flux_pin.resize (n_groups, std::vector<double> (n_pins, 0.0));
278  for (unsigned int i = 0; i < m_power_pin.size (); ++i)
279  {
280  m_norm_power_pin[i] = m_power_pin[i]/m_weight_pin[i]*m_scaling;
281  }
282  // Normalization (average flux per ass is equal to one (for fast group))
283  for (unsigned int g = 0; g < m_flux_pin.size (); ++g)
284  {
285  for (unsigned int i = 0; i < m_flux_pin[g].size (); ++i)
286  {
287  m_norm_flux_pin[g][i] = m_flux_pin[g][i]/m_weight_pin[i]*m_scaling;
288  }
289  }
290 
291  // Normalization (average power per ass is equal to one) ------------------
292  m_norm_power_ass.resize (n_pins, 0);
293  m_norm_flux_ass.resize (n_groups, std::vector<double> (n_ass, 0.0));
294  for (unsigned int i = 0; i < m_power_ass.size (); ++i)
295  {
296  m_norm_power_ass[i] = m_power_ass[i]/m_weight_ass[i]*m_scaling;

299  for (unsigned int g = 0; g < m_flux_ass.size (); ++g)

309  template <int dim>
310  double
311  PostProcessing<dim>::calculate_scaling (const std::vector<double> & vec)
312  {
313  // Normalization (average power per pin is equal to one)
314  double vec_sum = 0;
315  double vec_counter = 0;
316  for (unsigned int i = 0; i < vec.size (); ++i)
317  {
318  if (std::abs (vec[i]) > 1.e-10)
319  {
320  vec_sum += vec[i];
321  vec_counter += 1.0;
322  }
323  }
324  double scal = vec_counter / vec_sum;
325  return scal;
326  }
