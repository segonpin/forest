424  template <int dim>
425  void
426  MatrixFreeIntegrator<dim>::cell_matrix (Vector<double> &dst,
427  const Vector<double> &src,
428  const FEValuesBase<dim> &fe,
429  const double diff,
430  const double sigmar) const
431  {
432  for (unsigned int k = 0; k < fe.n_quadrature_points; ++k)
433  {
434  // Prepare constants
435  const double dx = fe.JxW (k);
436  // Evaluate trial function
437  Tensor<1, dim> dui;
438  double ui = 0;
439  for (unsigned int j = 0; j < fe.dofs_per_cell; ++j)
440  {
441  dui += fe.shape_grad (j, k) * src (j);
442  ui += fe.shape_value (j, k) * src (j);
443  }
444  // Multiply by coefficient and quadrature weight
445  dui *= dx * diff;
446  ui *= dx * sigmar;
447  // Multiply by test function
448  for (unsigned int i = 0; i < fe.dofs_per_cell; ++i)
449  {
450  Tensor<1, dim> dvi = fe.shape_grad (i, k);
451  double vi = fe.shape_value (i, k);
452  dst (i) += dvi * dui + vi * ui;
453  }
454  }
455  }

533  template <int dim>
534  void
535  MatrixFreeIntegrator<dim>::ip_matrix (Vector<double> & cell_dst,
536  Vector<double> & neighbor_dst,
537  const Vector<double> & cell_src,
538  const Vector<double> & neighbor_src,
539  const FEValuesBase<dim> &fe1,
540  const FEValuesBase<dim> &fe2,
541  const double penalty,
542  const double diff1,
543  const double diff2) const
544  {
545  for (unsigned int q = 0; q < fe1.n_quadrature_points; ++q)
546  {
547  // Prepare constants
548  const double dx = fe1.JxW (q);
549  const Tensor<1, dim> &n = fe1.normal_vector (q);
550  // Evaluate trial function
551  double ui = 0.;
552  double dnui = 0.;
553  double ue = 0.;
554  double dnue = 0.;
555  for (unsigned int j = 0; j < fe1.dofs_per_cell; ++j)
556  {
557  ui += fe1.shape_value (j, q) * cell_src (j);
558  dnui += n * fe1.shape_grad (j, q) * cell_src (j);
559  ue += fe2.shape_value (j, q) * neighbor_src (j);
560  dnue += n * fe2.shape_grad (j, q) * neighbor_src (j);
561  }
562  // Multiply by coefficient and quadrature weight
563  ui *= dx;
564  dnui *= diff1 * dx;
565  ue *= dx;
566  dnue *= diff2 * dx;
567  // Multiply by test function
568  for (unsigned int i = 0; i < fe1.dofs_per_cell; ++i)
569  {
570  const double vi = fe1.shape_value (i, q);
571  const double dnvi = diff1 * n * fe1.shape_grad (i, q);
572  const double ve = fe2.shape_value (i, q);
573  const double dnve = diff2 * n * fe2.shape_grad (i, q);
574  cell_dst (i) += -.5 * dnvi * ui - .5 * dnui * vi + penalty * ui * vi;
575  cell_dst (i) += .5 * dnvi * ue - .5 * vi * dnue - penalty * vi * ue;
576  neighbor_dst (i) += -.5 * dnve * ui + .5 * ve * dnui
577  - penalty * ve * ui;
578  neighbor_dst (i) += .5 * dnve * ue + .5 * ve * dnue + penalty * ve * ue;
579  }
580  }
