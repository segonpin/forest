396  template <int dim>
397  double
398  MatrixIntegrator<dim>::compute_penalty (const MeshWorker::DoFInfo<dim, dim,
399  double> &dinfo1,
400  const MeshWorker::DoFInfo<dim, dim,
401  double> &dinfo2,
402  const unsigned int deg1,
403  const unsigned int deg2,
404  const double coeff1,
405  const double coeff2) const
406  {
407  const unsigned int normal1 =
408  GeometryInfo<dim>::unit_normal_direction[dinfo1.face_number];
409  const unsigned int normal2 =
410  GeometryInfo<dim>::unit_normal_direction[dinfo2.face_number];
411  const unsigned int deg1sq = (deg1 == 0) ? 1 : deg1 * (deg1 + 1);
412  const unsigned int deg2sq = (deg2 == 0) ? 1 : deg2 * (deg2 + 1);
413 
414  double penalty1 = deg1sq / dinfo1.cell->extent_in_direction (normal1);
415  double penalty2 = deg2sq / dinfo2.cell->extent_in_direction (normal2);
416  if (dinfo1.cell->has_children () ^ dinfo2.cell->has_children ())
417  {
418  Assert(dinfo1.face == dinfo2.face, ExcInternalError());
419  Assert(dinfo1.face->has_children(), ExcInternalError());
420  penalty1 *= 2;
421  }
422  const double scal = 2.0;
423  const double penalty = scal * 0.5 * (coeff1 * penalty1 + coeff2 * penalty2);
424  return penalty;
425  }
