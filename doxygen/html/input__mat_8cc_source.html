433  void
434  InputMat::check ()
435  {
436  for (typename XS_map::iterator xs = m_xs.begin (); xs != m_xs.end (); ++xs)
437  {
438  // Transport and Diffusion cross sections -------------------------------
439 
440  // Check for nusigf and calculate it in case it is not provided.
441  if (xs->second.exist_nu && !xs->second.exist_sigf)
442  {
443  // in case it was provided, check consistency for given tolerance.
444  if (xs->second.exist_nusigf)
445  {
446  xs->second.check_nusigf ();
447  }
448  xs->second.calculate_nusigf ();
449  }
450 
451  // Normalize chi in case it was provided (should always be provided?)
452  if (xs->second.exist_chi)
453  {
454  xs->second.normalize_chi ();
455  }
456 
457  // Transport cross sections ---------------------------------------------
458 
459  // Check consistency for sigmat in case several data is provided
460  if (xs->second.exist_sigmas && xs->second.exist_sigmaa)
461  {
462  // Check consistency for sigmat in case it is provided
463  if (xs->second.exist_sigmat)
464  {
465  xs->second.check_sigmat ();
466  }
467  // Calculate sigmat, provided or not.
468  xs->second.calculate_sigmat ();
469  }
470  else if (xs->second.exist_sigmas && xs->second.exist_sigmat)
471  {
472  // sigmas and sigmat exist, but not sigmaa, so we calculate it
473  xs->second.calculate_sigmaa ();
474  }
475 
476  // Diffusion cross sections ---------------------------------------------
477 
478  if (xs->second.exist_sigmat && xs->second.exist_sigmas)
479  xs->second.calculate_diff ();
480 
481  if (xs->second.exist_sigmaa && xs->second.exist_sigmas)
482  xs->second.calculate_sigmar ();
483 
484 #ifdef DEBUG
485 
486  // Check if the problem has been completely defined for transport
487  bool problem_defined = false;
488  if (xs->second.exist_sigmat && xs->second.exist_sigmas
489  && xs->second.exist_chi && xs->second.exist_nusigf)
490  {
491  problem_defined = true;
492  }
493  // Check if the problem has been completely defined for diffusion
494  if (xs->second.exist_diff && xs->second.exist_sigmar
495  && xs->second.exist_sigmas && xs->second.exist_chi
496  && xs->second.exist_nusigf)
497  {
498  problem_defined = true;
499  }
500  Assert(problem_defined, dealii::ExcMessage("Problem not fully defined."));
501 #endif
502  }
