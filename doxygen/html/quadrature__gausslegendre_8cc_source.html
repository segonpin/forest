39  std::vector<double> &w, unsigned int n,
40  const double x1, const double x2)
41  {
42  // High precision is a good idea for this routine.
43  const double eps = 1.0e-14;
44 
45  // The roots are symmetric in the interval, so we only have
46  // to find half of them.
47  const long double xm = 0.5 * (x2 + x1);
48  const long double xl = 0.5 * (x2 - x1);
49 
50  // Loop over the desired roots.
51  unsigned int m = (n + 1) / 2;
52  for (unsigned int i = 1; i <= m; i++)
53  {
54  long double z = std::cos (Forest::pi * (i - 0.25) / (n + 0.5));
55  // Starting with the above approximation to the i'th root,
56  // we enter the main loop of refinement by Newton's method.
57  long double pp, p1, p3;
58  do
59  {
60  // Loop up the recurrence relation to get the
61  // Legendre polynomial evaluated at z.
62  long double p2;
63  p1 = 1.0;
64  p2 = 0.0;
65  for (unsigned int j = 0; j < n; j++)
66  {
67  p3 = p2;
68  p2 = p1;
69  p1 = ((2.0 * j + 1.0) * z * p2 - j * p3) / (j + 1);
70  }
71  // p1 is now the desired Legendre polynomial. We next
72  // compute pp, its derivative, by a standard relation
73  // involving also p2, the polynomial of one lower order.
74  pp = n * (z * p1 - p2) / (z * z - 1.0);
75  // Newton's method.
76  z = z - p1 / pp;
77  }
78  while (std::abs (p1 / pp) > eps);
79  // Scale the root to the desired interval, and put in its
80  // symmetric counterpart, (we invert the order to fit in
81  // a natural way with the boundary order for the discrete ordinates).
82  double x_aux1 = xm - xl * z;
83  double x_aux2 = xm + xl * z;
84 
85  x[n - i][0] = x_aux1;
86  x[i - 1][0] = x_aux2;
87  // Compute the weight and its symmetric counterpart.
88  double w_aux = 2.0 * xl / ((1.0 - z * z) * pp * pp);
89  w[n - i] = w_aux;
90  w[i - 1] = w_aux;
91  }
92  }
