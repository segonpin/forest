414  template <int dim>
415  void
416  QuadratureLevelSymType2<dim>::set_ordinates ()
417  {
418  std::vector<double> x_tmp (n_cosines);
419  std::vector<double> w_tmp (this->n_angles_per_octant);
420  std::vector<unsigned int> w_dist (this->n_angles_per_octant);
421 
422  set_ind_quad (x_tmp, w_tmp, this->order);
423  get_weight_dist (w_dist);
424 
425  unsigned int k = 0;
426  for (unsigned int i = 0; i < n_cosines; ++i)
427  {
428  for (unsigned int j = 0; j < i + 1; ++j, ++k)
429  {
430  this->xx[k][0] = x_tmp[j];
431  this->xx[k][1] = x_tmp[i - j];
432  this->xx[k][2] = std::sqrt (
433  1.0 - this->xx[k][0] * this->xx[k][0]
434  - this->xx[k][1] * this->xx[k][1]);
435  this->ww[k] = w_tmp[w_dist[k] - 1];
436  }
437  }
438 
439  w_dist.clear ();
440  x_tmp.clear ();
441  w_tmp.clear ();
442  }
