82  template <int dim>
83  void
84  Homogenization<dim>::cell_weights ()
85  {
86  /* We need some constants */
87  const unsigned int n_groups = mp_state.get_n_groups ();
88  const unsigned int n_cells = geom.m_tria.n_active_cells ();
89 
90  /* Initialize all that is needed to iterate over dofs and cells */
91  unsigned int fe_degree = data.mp_settings.get_fe_degree ();
92  QGauss<dim> quad_formula (fe_degree + 1);
93  FEValues<dim> fe_values (fe, quad_formula,
94  update_values | update_quadrature_points | update_volume_elements
95  | update_JxW_values);
96 
97  const unsigned int n_q = quad_formula.size ();
98  std::vector<unsigned int> global_dof_ind (fe.dofs_per_cell);
99 
100  /* Assuming the cross sections is constant per cell, we need the
101  * integral of the flux per cell, that will be used for the weighting
102  * of the cross sections */
103  cell_w.resize (n_cells, std::vector<double> (n_groups, 0.));
104 
105  /* For shorter access to the scalar flux */
106  const SnVector & flux = mp_state.m_scalar_flux;
107 
108  /* Run over all the cells to calculate each weight */
109  c_iter cell_it = dof_handler.begin_active (), endc = dof_handler.end ();
110  for (; cell_it != endc; ++cell_it)
111  {
112  fe_values.reinit (cell_it);
113  cell_it->get_dof_indices (global_dof_ind);
114  const unsigned int cell = cell_it->user_index ();
115 
116  /* for every degree of freedom in this cell... */
117  for (unsigned int i = 0; i < fe.dofs_per_cell; ++i)
118  {
119  /* ...we calculate the integral of the associated base function, */
120  double integral = 0;
121  for (unsigned int q = 0; q < n_q; ++q)
122  integral += fe_values.shape_value (i, q) * fe_values.JxW (q);
123 
124  /* and then we use it for the density of every group. */
125  for (unsigned int g = 0; g < n_groups; ++g)
126  cell_w[cell][g] += flux.m_data[g][global_dof_ind[i]] * integral;
127  }
128  }
129  }
